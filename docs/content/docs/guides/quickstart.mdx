import { Callout } from 'fumadocs-ui/components/callout';
import { Steps, Step } from 'fumadocs-ui/components/steps';
import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Callout type="info">
No account, email, or API key required. Everything runs locally.
</Callout>

<Steps>
<Step>
### Install

<Tabs items={['npm', 'cargo']}>
<Tab value="npm">
```bash
npm install -g treeship
```
</Tab>
<Tab value="cargo">
```bash
cargo install treeship-cli
```
</Tab>
</Tabs>
</Step>
<Step>
### Initialize

```bash
treeship init
```

```
✓ Treeship initialized
  name: my-workspace
  key:  key_9f8e7d6c (ed25519, encrypted at rest)
```
</Step>
<Step>
### Wrap a command

```bash
treeship wrap -- npm test
```

```
✓ action attested
  id:      art_f7e6d5c4
  actor:   human://alice
  action:  npm test
  exit:    0
  elapsed: 1.4s
```
</Step>
<Step>
### Verify

```bash
treeship verify last
```

```
✓ verified
  chain:  1 artifact, intact
  signed: human://alice, 2026-03-26T21:00:00Z
```
</Step>
<Step>
### Share

```bash
treeship hub attach
# Visit the URL shown, enter the code, click Approve

treeship hub push last
# -> https://treeship.dev/verify/art_f7e6d5c4
```

Anyone with that URL can verify the full chain. No account or install needed.
</Step>
</Steps>
